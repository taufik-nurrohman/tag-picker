Test native HTML5 custom element. View the page source!

Usage

<!DOCTYPE html>
<html dir="ltr">
  <head>
    <meta charset="utf-8">
    <link href="./index.min.css" rel="stylesheet">
  </head>
  <body>
    <p><tag-picker name="tags" value="foo, bar, baz"></tag-picker></p>
    <script src="./index.min.js"></script>
    <script src="./index/element.min.js"></script>
    <script>
    // Get tag picker element
    const picker = document.querySelector('tag-picker');
    // Make tag picker disabled
    picker.disabled = true;
    // Make tag picker freezed
    picker.readOnly = true;
    // Set tag picker value
    picker.value = 'foo, bar';
    // Get tag picker value as string
    console.log(picker.value);
    </script>
  </body>
</html>
